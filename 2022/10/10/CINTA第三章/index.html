9.编写Python语言程序完成欧拉Phi函数的计算，即输入正整数n，计算并返回φ(n).

import math

def gcd(a,b):
    if b==0:
        return a
    else:
        return gcd(b,a%b)

def jud(num):#判断素数
    a=1
    for i in range(2,math.ceil(math.sqrt(num))):
        if num%i==0:
            a=0
    if a==1:
        return 1
    else:
        return 0

def Phi(n):
    if jud(n)==1:
        return n-1
    else:
        count=0
        for i in range(1,n):
            if gcd(n,i)==1:
                count=count+1
        return count
